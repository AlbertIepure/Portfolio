1function Flinelikeness = linelikeness(I,theta,d)
  2% parfor j=1:10
  3%     linelikeness(j) = linelikeness3_mex(double(I),theta,j);
  4% end
  5% 
  6% Flinelikeness1 = sum(linelikeness(1:10));
  7
  8n=16;
  9
 10PDd=zeros(n,n,8); % Co-occurrence matrix structure with 8 directions  
 11[h,w]=size(I);
 12
 13for i=1:h
 14    for j=1:w
 15        for k1=1:n
 16            for k2=1:n
 17                % We construct for each one of the eight possible
 18                % directions a co-occurence matrix whose element PDd(i,j)
 19                % is defined as the relative frequency with which two
 20                % neighboring cells separated by a distance d along the
 21                % edge direction occur on the image, one with the direction
 22                % code i and the other with the direction code j
 23                
 24                if (theta(i,j)>=(2*(k1-1)*pi/2/n) && theta(i,j)<((2*(k1-1)+1)*pi/2/n))
 25                    %Down direction
 26                    if i+d <=h
 27                        if (theta(i+d,j)>=(2*(k2-1)*pi/2/n) && theta(i+d,j)<((2*(k2-1)+1)*pi/2/n))
 28                            PDd(k1,k2,1)=PDd(k1,k2,1)+1;
 29                        end
 30                    end
 31                    
 32                    %Up Direction
 33                    if i-d >= 1
 34                        if (theta(i-d,j)>=(2*(k2-1)*pi/2/n) && theta(i-d,j)<((2*(k2-1)+1)*pi/2/n))
 35                            PDd(k1,k2,2)=PDd(k1,k2,2)+1;
 36                        end
 37                    end
 38                    
 39                    if j+d <= w
 40                        %Right direction
 41                        if (theta(i,j+d)>=(2*(k2-1)*pi/2/n) && theta(i,j+d)<((2*(k2-1)+1)*pi/2/n))
 42                            PDd(k1,k2,3)=PDd(k1,k2,3)+1;
 43                        end
 44                        
 45                        if i+d <= h
 46                            %Lower right direction
 47                            if (theta(i+d,j+d)>=(2*(k2-1)*pi/2/n) && theta(i+d,j+d)<((2*(k2-1)+1)*pi/2/n))
 48                                PDd(k1,k2,5)=PDd(k1,k2,5)+1;
 49                            end
 50                        end
 51                        
 52                        if i-d >= 1
 53                            %Upper right direction
 54                            if (theta(i-d,j+d)>=(2*(k2-1)*pi/2/n) && theta(i-d,j+d)<((2*(k2-1)+1)*pi/2/n))
 55                                PDd(k1,k2,6)=PDd(k1,k2,6)+1;
 56                            end
 57                        end
 58                    end
 59                    
 60                    if j-d >= 1
 61                        %Left direction
 62                        if (theta(i,j-d)>=(2*(k2-1)*pi/2/n) && theta(i,j-d)<((2*(k2-1)+1)*pi/2/n))
 63                            PDd(k1,k2,4)=PDd(k1,k2,4)+1;
 64                        end
 65                        
 66                        if i+d <= h
 67                            %Lower left direction
 68                            if (theta(i+d,j-d)>=(2*(k2-1)*pi/2/n) && theta(i+d,j-d)<((2*(k2-1)+1)*pi/2/n))
 69                                PDd(k1,k2,7)=PDd(k1,k2,7)+1;
 70                            end
 71                        end
 72                        
 73                        if i-d >= 1
 74                            % Upper left direction
 75                            if (theta(i-d,j-d)>=(2*(k2-1)*pi/2/n) && theta(i-d,j-d)<((2*(k2-1)+1)*pi/2/n))
 76                                PDd(k1,k2,8)=PDd(k1,k2,8)+1;
 77                            end
 78                        end
 79                    end
 80                end
 81            end
 82        end
 83    end
 84end
 85
 86numerator=zeros(1,8);
 87denominator=zeros(1,8);
 88
 89for k=1:8
 90    denominator(k) = sum(sum(PDd(:,:,k)));
 91    for i=1:n
 92        for j=1:n
 93            numerator(k) = numerator(k) + PDd(i,j,k)*cos((i-j)*2*pi/n);
 94        end
 95    end
 96end
 97
 98l=numerator./denominator; % vector containing for each 45 deg direction the linelikeness value
 99
100Flinelikeness = sum(l);
